## 💡 筆者のコメント (Insight)

## 📊 証明：Before & After

### ❌ Before (入力：レガシーなコールバック地獄)

```javascript
function getUserData(userId, callback) {
  db.query("SELECT * FROM users WHERE id = " + userId, function (err, user) {
    if (err) {
      callback(err);
    } else {
      db.query(
        "SELECT * FROM posts WHERE author_id = " + user.id,
        function (err, posts) {
          if (err) {
            callback(err);
          } else {
            user.posts = posts;
            callback(null, user);
          }
        },
      );
    }
  });
}
```

### ✅ After (結果：モダンな非同期処理とSQLインジェクション対策)

```javascript
/**
 * ユーザーデータと関連する投稿を取得します。
 * @param {number} userId - ユーザーID
 * @returns {Promise<Object>} 投稿情報を含むユーザーオブジェクト
 */
async function getUserData(userId) {
  try {
    // プレースホルダを使用した安全なクエリ実行 (SQLインジェクション対策)
    const [user] = await db.execute("SELECT * FROM users WHERE id = ?", [
      userId,
    ]);

    if (!user) {
      throw new Error("User not found");
    }

    // 関連する投稿データの取得
    const posts = await db.execute("SELECT * FROM posts WHERE author_id = ?", [
      user.id,
    ]);

    return {
      ...user,
      posts,
    };
  } catch (error) {
    console.error("Failed to fetch user data:", error);
    throw error; // エラーハンドリングの呼び出し元への移譲
  }
}
```
