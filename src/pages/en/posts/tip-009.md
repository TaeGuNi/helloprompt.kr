  // 1. Happy Path: Normal division
  test("10 divided by 2 should be 5", () => {
    expect(divide(10, 2)).toBe(5);
  });

  // 2. Edge Case: Negative division
  test("dividing negative numbers should handle signs correctly", () => {
    expect(divide(-10, 2)).toBe(-5);
    expect(divide(-10, -2)).toBe(5);
  });

  // 3. Exception Case: Division by zero
  test("dividing by 0 should throw an error", () => {
    expect(() => divide(10, 0)).toThrow("Cannot divide by zero.");
  });

  // 4. Edge Case: Decimal precision
  test("division result should be accurate when it's a decimal", () => {
    expect(divide(5, 2)).toBe(2.5);
  });
});
