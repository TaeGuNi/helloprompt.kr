```python
# プロンプトを通じて生成されたコア・ハイブリッドパイプラインコード (例)
import cirq
import sympy
import tensorflow as tf
import tensorflow_quantum as tfq

# 1. 量子ビット(Qubit)およびパラメータ化された量子回路(PQC)の構成
qubit = cirq.GridQubit(0, 0)
theta = sympy.Symbol('theta')
circuit = cirq.Circuit(cirq.rx(theta)(qubit))

# 2. 量子レイヤー(PQC)を古典的なKerasモデルにシームレスに統合

    tf.keras.layers.Dense(64, activation='relu'),
    # 古典的なテンソルデータを入力として受け取り、量子回路を経て期待値を出力
    tfq.layers.PQC(circuit, cirq.Z(qubit)),
    tf.keras.layers.Dense(1, activation='sigmoid')
])

model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=0.01),
              loss='binary_crossentropy')
```
