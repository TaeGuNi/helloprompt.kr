## 🙋 常见问题 (FAQ)

- **Q: 在不支持 WebGPU 的旧款浏览器或老旧设备上，真的完全无法运行吗？**
  - A: 是的，底层的 WebGPU 硬件加速是必要前提。但在实际的商业项目落地时，业界标准的做法正如我们在提示词中所约束的那样：在应用初始化阶段首先检测当前设备的 WebGPU 兼容性。一旦发现不支持，则触发优雅降级（Graceful Degradation），无缝回退并调用云端的大模型 API（如 OpenAI 等），从而构建一套稳如磐石的“云端-本地混合架构（Hybrid Architecture）”。

- **Q: 动辄数 GB 的模型下载体积，难道不会严重影响用户的访问体验并导致高跳出率吗？**

- **Q: 这种本地推理的文本生成速度（TPS，每秒生成 Token 数），真的能达到实际业务中日常使用的标准吗？**
  - A: 绝对可以。在搭载 Apple M2/M3 系列芯片的 Mac，或是配备了较新独立显卡的 Windows PC 环境下，其实际文本生成速度能够轻松飙升至惊人的每秒 30~50 个 Token 甚至更高。你所体验到的响应速度和文字流输出的流畅度，将丝毫不逊色于那些昂贵的顶级云端 API。

```tsx
// useWebLLM.ts (由提示词生成的生产级代码片段)
import { useState, useEffect } from "react";
import { CreateMLCEngine } from "@mlc-ai/web-llm";

export const useWebLLM = (modelId = "Llama-3-8B-Instruct-q4f32_1") => {
  const [progress, setProgress] = useState(0);
  const [engine, setEngine] = useState<any>(null);
  const [isReady, setIsReady] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const initEngine = async () => {
      if (!navigator.gpu) {
        setError("当前浏览器不支持 WebGPU硬件加速。");
        return;
      }
      try {
        const mlcEngine = await CreateMLCEngine(modelId, {
          initProgressCallback: (report) => {
            setProgress(Math.round(report.progress * 100));
          },
        });
        setEngine(mlcEngine);
        setIsReady(true);
      } catch (err) {
        setError("模型加载过程中发生错误，请重试。");
      }
    };
    initEngine();
  }, [modelId]);

  return { engine, isReady, progress, error };
};
```
