## 📊 証明: Before & After

### ❌ Before (入力コード)

```javascript
function divide(a, b) {
  if (b === 0) throw new Error("0で割ることはできません。");
  return a / b;
}
```

### ✅ After (結果)

```javascript
describe("divide 関数テスト", () => {
  // 1. 正常動作テスト
  test("10を2で割ると5になるべき", () => {
    expect(divide(10, 2)).toBe(5);
  });

  // 2. 例外処理テスト
  test("0で割るとエラーが発生するべき", () => {
    expect(() => divide(10, 0)).toThrow("0で割ることはできません。");
  });

  // 3. 境界値/特殊値テスト
  test("負の数を割ると負数になるべき", () => {
    expect(divide(-10, 2)).toBe(-5);
  });
});
```
