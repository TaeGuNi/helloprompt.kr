## 📊 证明: Before & After

### ❌ Before (输入: 简单的计算函数)

```javascript
function divide(a, b) {
  if (b === 0) throw new Error("不能除以0。");
  return a / b;
}
```

<br>

### ✅ After (结果)

```javascript
// 使用 Jest 框架示例

describe("divide 函数测试", () => {
  // 1. Happy Path: 正常的除法
  test("10除以2应该得5", () => {
    expect(divide(10, 2)).toBe(5);
  });

  // 2. Edge Case: 负数除法
  test("除以负数时符号应该被正确处理", () => {
    expect(divide(-10, 2)).toBe(-5);
    expect(divide(-10, -2)).toBe(5);
  });

  // 3. Exception Case: 除以0
  test("除以0时应该抛出错误", () => {
    expect(() => divide(10, 0)).toThrow("不能除以0。");
  });

  // 4. Edge Case: 小数点处理
  test("除法结果为小数时应该准确", () => {
    expect(divide(5, 2)).toBe(2.5);
  });
});
```
