## 📊 证明: Before & After

### ❌ Before (输入: 简单计算函数)

```javascript
function divide(a, b) {
  if (b === 0) throw new Error("不能除以 0。");
  return a / b;
}
```

<br>

### ✅ After (结果)

```javascript
// Jest 框架使用示例

describe("divide 函数测试", () => {
  // 1. Happy Path: 正常除法
  test("10 除以 2 应该得到 5", () => {
    expect(divide(10, 2)).toBe(5);
  });

  // 2. Edge Case: 负数除法
  test("除以负数时符号应该被正确处理", () => {
    expect(divide(-10, 2)).toBe(-5);
    expect(divide(-10, -2)).toBe(5);
  });

  // 3. Exception Case: 除以 0
  test("除以 0 应该抛出错误", () => {
    expect(() => divide(10, 0)).toThrow("不能除以 0。");
  });

  // 4. Edge Case: 小数点处理
  test("除法结果为小数时应该准确", () => {
    expect(divide(5, 2)).toBe(2.5);
  });
});
```
