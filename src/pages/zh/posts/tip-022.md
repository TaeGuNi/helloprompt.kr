## 📊 证明: Before & After

### ❌ Before (输入代码)

```javascript
function divide(a, b) {
  if (b === 0) throw new Error("不能除以0。");
  return a / b;
}
```

### ✅ After (结果)

```javascript
describe("divide 函数测试", () => {
  // 1. 正常动作测试
  test("10除以2应该得5", () => {
    expect(divide(10, 2)).toBe(5);
  });

  // 2. 异常处理测试
  test("除以0应该发生错误", () => {
    expect(() => divide(10, 0)).toThrow("不能除以0。");
  });

  // 3. 边界值/特殊值测试
  test("除以负数应该变成负数", () => {
    expect(divide(-10, 2)).toBe(-5);
  });
});
```
